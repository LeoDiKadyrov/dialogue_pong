```js
// Extracted rate limit logic to test in isolation
function createRateLimiter(limitMs) {
  const lastTime = new Map();
  return {
    isAllowed(socketId) {
      const now = Date.now();
      const last = lastTime.get(socketId) ?? 0;
      if (now - last < limitMs) return false;
      lastTime.set(socketId, now);
      return true;
    },
    delete(socketId) {
      lastTime.delete(socketId);
    },
  };
}
```

```jsx
import React, { useState, useEffect, useRef } from 'react';

/**
 * SessionTimer — Shows elapsed time since game started
 * Replaces ScoreBoard: no competitive score, just "how long have we been talking"
 */
function SessionTimer() {
  const [elapsed, setElapsed] = useState(0); // seconds
  const intervalRef = useRef(null);

  useEffect(() => {
    intervalRef.current = setInterval(() => {
      setElapsed((s) => s + 1);
    }, 1000);
    return () => clearInterval(intervalRef.current);
  }, []);

  const hours = Math.floor(elapsed / 3600);
  const minutes = Math.floor((elapsed % 3600) / 60);
  const seconds = elapsed % 60;

  const pad = (n) => String(n).padStart(2, '0');
  const display = hours > 0
    ? `${pad(hours)}:${pad(minutes)}:${pad(seconds)}`
    : `${pad(minutes)}:${pad(seconds)}`;

  return (
    <div className="session-timer" aria-label="Session duration">
      <span className="session-timer-label">session</span>
      <span className="session-timer-value">{display}</span>
    </div>
  );
}

export default SessionTimer;
```

```js
// Leet speak substitution map
const LEET_MAP = {
  '4': 'a', '@': 'a',
  '3': 'e',
  '1': 'i', '!': 'i',
  '0': 'o',
  '5': 's', '$': 's',
  '7': 't',
};

/**
 * Normalize text: remove accents, apply leet substitutions — for filter comparison only.
 * Does NOT modify the original text, only used to detect blocked words.
 * @param {string} text
 * @returns {string}
 */
function normalizeForFilter(text) {
  // Remove accents (NFD decomposes é → e + combining accent, then strip combining chars)
  const noAccents = text.normalize('NFD').replace(/[\u0300-\u036f]/g, '');
  // Apply leet substitutions
  return noAccents.replace(/[4@3!105$7]/g, (c) => LEET_MAP[c] ?? c).toLowerCase();
}

/**
 * Filter profanity from a message
 * @param {string} text - Raw message text
 * @returns {string} Filtered message with bad words replaced by ***
 */
export function filterMessage(text) {
  if (!text || typeof text !== 'string') return text;

  const normalized = normalizeForFilter(text);
  let result = text;

  for (const word of BLOCKED_WORDS) {
    // Test against normalized version, replace in original
    const regex = new RegExp(`\\b${word}\\b`, 'gi');
    if (regex.test(normalized)) {
      // Find matching spans in normalized text and replace same positions in original
      // Simpler approach: rebuild result replacing char-by-char matches
      const normRegex = new RegExp(`\\b${word}\\b`, 'gi');
      let match;
      const replacements = [];
      while ((match = normRegex.exec(normalized)) !== null) {
        replacements.push({ start: match.index, end: match.index + match[0].length });
      }
      // Replace from end to avoid offset shifts
      for (let i = replacements.length - 1; i >= 0; i--) {
        const { start, end } = replacements[i];
        result = result.slice(0, start) + '***' + result.slice(end);
      }
    }
  }
  return result;
}
```
